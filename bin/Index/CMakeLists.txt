set(exe_name "mx-index")

    "IndexCompileJob.cpp"
    "IndexCompileJob.h"
    "Main.cpp"
    "PrintTokenGraph.cpp"
    "PrintTokenGraph.h"
    "TokenizeFile.cpp"
    "TokenizeFile.h"
