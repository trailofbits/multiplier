set(exe_name "mx-import")

add_executable("${exe_name}"
  "Log.h"
  
  "Parser.h"
  "Parser.cpp"
  
  "Importer.h"
  "Importer.cpp"
  
  "Main.cpp"
)

set_target_properties("${exe_name}"
  PROPERTIES
    LINKER_LANGUAGE
      CXX)

  PRIVATE
    "util"
)

if(IRENE_ENABLE_INSTALL)
  install(
    TARGETS
      "${exe_name}"
    EXPORT
      "${PROJECT_NAME}Targets"
    RUNTIME
      DESTINATION
        "${CMAKE_INSTALL_BINDIR}"
  )
endif()
